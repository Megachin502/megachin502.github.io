Animating SVG Example

This is the code starting the animations:

slide = function() {
  final = {
    'width': '15',
    'height': '15',
    'transform': 't130,0'
  }
  sq.animate(final, 2000)
}
spin = function() {
  initial = {
    'transform': 'r0'
  }
  final = {
    'transform': 'r360'
  }
  sq.attr(initial)
  sq.animate(final, 2000)
}
